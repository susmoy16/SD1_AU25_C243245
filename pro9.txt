import java.util.*;

public class Problem9 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int total = 0;
        int counter = 1;

        while (counter <= n) {
            total += counter;
            counter++;
        }

        System.out.println(total);
    }
}
