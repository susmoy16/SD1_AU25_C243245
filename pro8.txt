import java.util.*;

public class Problem8 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        long result = 1; // different variable name and type

        int counter = n;
        while (counter > 0) {
            result *= counter;
            counter--;
        }

        System.out.println(result);
    }
}
